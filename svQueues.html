  initial begin

    $display("Push apricot, size=%0d fruits=%p", fruits.size(), fruits);
    $display("Pop %s, size=%0d fruits=%p", fruits.pop_back(), fruits.size(), fruits);
    fruits.push_back("plum");
    $display("Push plum, size=%0d fruits=%p", fruits.size(), fruits);
  end
endmodule
